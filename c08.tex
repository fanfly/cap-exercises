\begin{problem}
  \item[8.] 若一元二次方程式 $(x + 3)^2 = 2$ 的兩根為 $a$、$b$，且 $a > b$，則 $b$ 之值為何？
  \begin{choices}
    \item $3+\sqrt{2}$
    \item $3-\sqrt{2}$
    \item $-3+\sqrt{2}$
    \item $-3-\sqrt{2}$
  \end{choices}
\end{problem}
