\begin{problem}
  \item[6.] 箱子內有 5 顆球，其種類與數量如表（一）所示。若箱中每顆球被抽出的機會相等，則從箱子中抽出紅球的機率為何？
  \begin{table}[ht]
    \centering
    \renewcommand{\arraystretch}{1.2}
    \vspace*{-1ex}
    \caption*{表（一）}
    \vspace*{-1ex}
    \begin{tabular}{|c|c|c|c|}
      \hline
      種類 & 黑球 & 紅球 & 藍球 \\ \hline
      數量（顆） & 2 & 2 & 1 \\ \hline
    \end{tabular}
    \vspace*{-2ex}
  \end{table}
  \begin{choices}
    \item $\fraction{1}{5}$
    \item $\fraction{2}{5}$
    \item $\fraction{3}{5}$
    \item $\fraction{4}{5}$
  \end{choices}
\end{problem}
