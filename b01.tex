\begin{problem}
  \item[1.] 已知甲、乙兩種粒子皆為電中性的原子，其原子序與質量數如表（一）所示。關於這兩種粒子的敘述，下列敘述何者正確？
  \begin{table}[ht]
    \centering
    \renewcommand{\arraystretch}{1.2}
    \vspace*{-1ex}
    \caption*{表（一）}
    \vspace*{-1ex}
    \begin{tabular}{|c|c|c|}
      \hline
      粒子 & 原子序 & 質量數 \\ \hline
      甲 & 1 & 1 \\ \hline
      乙 & 2 & 4 \\ \hline
    \end{tabular}
  \end{table}
  \begin{choices}
    \item 甲粒子不含質子
    \item 甲粒子不含中子
    \item 乙粒子含有 4 個質子
    \item 乙粒子含有 4 個中子
  \end{choices}
\end{problem}
