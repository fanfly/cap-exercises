\begin{problem}
  \item[2.] 已知甲、乙兩種粒子皆為單原子離子，其電子數與質子數如表（二）所示。關於這兩種粒子的敘述，下列何者正確？
  \begin{table}[ht]
    \centering
    \renewcommand{\arraystretch}{1.2}
    \vspace*{-1ex}
    \caption*{表（二）}
    \vspace*{-1ex}
    \begin{tabular}{|c|c|c|}
      \hline
      粒子 & 電子數 & 質子數 \\ \hline
      甲 & 18 & 17 \\ \hline
      乙 & 10 & 11 \\ \hline
    \end{tabular}
  \end{table}
  \begin{choices}
    \item 甲粒子的原子序為 17
    \item 乙粒子的原子序為 10
    \item 甲粒子是陽離子
    \item 乙粒子是陰離子
  \end{choices}
\end{problem}
