\begin{problem}
  \item[9.] 若正整數 $a$ 與 420 的最大公因數為 35，則下列敘述何者正確？
  \begin{choices}
    \item 20 可能是 $a$ 的因數，且 25 也可能是 $a$ 的因數
    \item 20 可能是 $a$ 的因數，但 25 不可能是 $a$ 的因數
    \item 20 不可能是 $a$ 的因數，但 25 可能是 $a$ 的因數
    \item 20 不可能是 $a$ 的因數，且 25 也不可能是 $a$ 的因數
  \end{choices}
\end{problem}
