\begin{problem}
  \item[8.] 有一等比數列，其第一項為 $10^3$，第二項為 $10^2$，第三項為 $10^1$。關於此等比數列，下列敘述何者正確？
  \begin{choices}
    \item 第四項為 $0$
    \item 第五項為 $0.001$
    \item 第六項為 $0.01$
    \item 第七項為 $0.1$
  \end{choices}
\end{problem}
