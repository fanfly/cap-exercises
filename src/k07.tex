\begin{problem}
  \item[7.] 有一等比數列，其第一項為 $1$，第二項為 $10$，第三項為 $100$。關於此等比數列，下列敘述何者正確？
  \begin{choices}
    \item 第四項為 $10^3$
    \item 第五項為 $10^3$
    \item 第六項為 $10^7$
    \item 第七項為 $10^7$
  \end{choices}
\end{problem}
