\begin{problem}[widest=10]
  \item[10.] 若 $A = 2^2 \times 3^3 \times 5^5$，則下列何者是 $A$ 的因數？
  \begin{choices}
    \item $2^6 \times 3$
    \item $2 \times 5^6$
    \item $3^3 \times 5^4$
    \item $2^2 \times 3^4 \times 5$
  \end{choices}
\end{problem}
