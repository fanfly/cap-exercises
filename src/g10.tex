\begin{problem}[widest=10]
  \item[10.] 丙烷（\ch{C3H8}）燃燒的反應式為：
  \begin{equation*}
    \ch{C3H8 + 5 O2 -> 3 CO2 + 4 H2O}
  \end{equation*}
  已知 \ch{H}、\ch{C}、\ch{O} 的原子量分別為 1、12、16。當 \qty{88}{g} 的丙烷完全燃燒時，下列敘述何者正確？
  \begin{choices}
    \item 會消耗 5 莫耳的氧氣
    \item 會生成 12 莫耳的二氧化碳
    \item 會消耗 \qty{320}{g} 的氧氣
    \item 會生成 \qty{72}{g} 的水
  \end{choices}
\end{problem}
