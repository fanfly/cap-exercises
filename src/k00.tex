\begin{problem}
  \item[A.] 10 的冪次：
  \begin{align*}
    10^2 &= 100 \\
    10^1 &= 10 \\
    10^0 &= 1 \\
    10^{-1} &= 0.1 \\
    10^{-2} &= 0.01
  \end{align*}
  \item[B.] 指數運算的例子：
  \begin{align*}
    2^5 \times 3^5 &= 6^5 \\
    2^5 \times 2^4 &= 2^9 \\
    2^9 \div 2^4 &= 2^5 \\
    (2^4)^3 &= 2^{12}
  \end{align*}
\end{problem}
