\begin{problem}
  \item[4.] 一個箱子內裝有三顆球，分別標記號碼 1、2、3。以從箱中每次取一球後放回的方式取球五次後，結果如表（一）所示。關於這五次取球的結果，下列敘述何者正確？
  \begin{table}[ht]
    \centering
    \renewcommand{\arraystretch}{1.2}
    \vspace*{-1ex}
    \caption*{表（一）}
    \vspace*{-1ex}
    \begin{tabular}{|c|c|c|c|c|c|}
      \hline
      次數 & 一 & 二 & 三 & 四 & 五 \\ \hline
      號碼 & 2 & 3 & 1 & 3 & 2 \\ \hline
    \end{tabular}
    \vspace*{-2ex}
  \end{table}
  \begin{choices}
    \item 取出號碼的平均數為 1.8
    \item 取出號碼的平均數為 2.4
    \item 取出號碼的中位數為 1
    \item 取出號碼的中位數為 2
  \end{choices}
\end{problem}
